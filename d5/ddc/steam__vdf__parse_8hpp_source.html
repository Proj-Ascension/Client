76 inline pt::ptree parseSection(std::istream& input, bool root, std::string name)
77 {
78  pt::ptree section;
79 
80  while (true)
81  {
82  uint8_t valueType = input.get();
83 
84  if (valueType == 0x08)
85  {
86  if (root)
87  {
88  input.get();
89  }
90  break;
91  }
92 
93  std::string key = readString(input);
94 
95  switch (valueType)
96  {
97  case 0x00: // None
98  {
99  section.add_child(key, parseSection(input, false, key));
100  break;
101  }
102  case 0x01: // string
103  {
104  section.put(key, readString(input));
105  break;
106  }
107  case 0x02: // int
108  {
109  section.put(key, read32_le(input));
110  break;
111  }
112  case 0x03: // float
113  {
114  std::cerr << "Floats are not implemented." << std::endl;
115  break;
116  }
117  case 0x04: // ptr
118  {
119  std::cerr << "Pointers are not implemented." << std::endl;
120  break;
121  }
122  case 0x05: // Wide string
123  {
124  std::cerr << "Wide strings are not implemented." << std::endl;
125  break;
126  }
127  case 0x06: // Colour
128  {
129  std::cerr << "Colours are not implemented." << std::endl;
130  break;
131  }
132  case 0x07: // 64 bit int
133  {
134  section.put(key, read64_le(input));
